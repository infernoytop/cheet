a) Bank Account Management System - Class and Object

#include <iostream>
using namespace std;

class BankAccount {
private:
    string name;
    int accountNumber;
    float balance;

public:
    void createAccount(string accName, int accNum, float initialBal) {
        name = accName;
        accountNumber = accNum;
        balance = initialBal;
    }

    void deposit(float amount) {
        balance += amount;
    }

    void withdraw(float amount) {
        if (amount <= balance)
            balance -= amount;
        else
            cout << "Insufficient balance!" << endl;
    }

    void display() {
        cout << "\nAccount Holder: " << name << endl;
        cout << "Account Number: " << accountNumber << endl;
        cout << "Balance: " << balance << endl;
    }
};

int main() {
    BankAccount acc;
    acc.createAccount("Pooja", 12345, 1000.0);
    acc.deposit(500.0);
    acc.withdraw(200.0);
    acc.display();
    return 0;
}

b) Area Calculator - Function Overloading

#include <iostream>
using namespace std;

class Area {
public:
    // Area of Circle
    float area(float radius) {
        return 3.14 * radius * radius;
    }

    // Area of Rectangle
    float area(float length, float breadth) {
        return length * breadth;
    }

    // Area of Triangle
    float area(int base, int height) {
        return 0.5 * base * height;
    }
};

int main() {
    Area a;
    float circleRadius = 5;
    float rectLength = 6, rectBreadth = 4;
    int triBase = 3, triHeight = 7;

    cout << "Area of Circle: " << a.area(circleRadius) << endl;
    cout << "Area of Rectangle: " << a.area(rectLength, rectBreadth) << endl;
    cout << "Area of Triangle: " << a.area(triBase, triHeight) << endl;

    return 0;
}

d) Number Swapping - Reference Arguments

#include <iostream>
using namespace std;

void swapNumbers(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
    int x = 10, y = 20;

    cout << "Before Swapping:\n";
    cout << "x = " << x << ", y = " << y << endl;

    swapNumbers(x, y);

    cout << "\nAfter Swapping:\n";
    cout << "x = " << x << ", y = " << y << endl;

    return 0;
}

e) Arithmetic Operations - Interface Implementation

#include <iostream>
using namespace std;

class Arithmetic {
private:
    int a, b;

public:
    void setValues(int x, int y) {
        a = x;
        b = y;
    }

    int add() {
        return a + b;
    }

    int subtract() {
        return a - b;
    }

    int multiply() {
        return a * b;
    }

    float divide() {
        if (b != 0)
            return (float)a / b;
        else
            return 0; // handle divide-by-zero
    }
};

int main() {
    Arithmetic calc;
    calc.setValues(20, 5);

    cout << "Addition: " << calc.add() << endl;
    cout << "Subtraction: " << calc.subtract() << endl;
    cout << "Multiplication: " << calc.multiply() << endl;
    cout << "Division: " << calc.divide() << endl;

    return 0;
}

2) Prime Number Check - While Loop

#include <iostream>
using namespace std;

int main() {
    int n, i = 2, flag = 0;
    cout << "Enter a number: ";
    cin >> n;

    while (i < n) {
        if (n % i == 0) {
            flag = 1;
            break;
        }
        i++;
    }

    if (n <= 1) {
        cout << n << " is NOT a Prime number." << endl;
    }
    else if (flag == 0) {
        cout << n << " is a Prime number." << endl;
    }
    else {
        cout << n << " is NOT a Prime number." << endl;
    }

    return 0;
}

3) Fibonacci Series - Do-While Loop

#include <iostream>
using namespace std;

int main() {
    int n, a = 0, b = 1, next, i = 1;
    cout << "Enter the number of terms: ";
    cin >> n;

    cout << "Fibonacci Series: ";
    do {
        cout << a << " ";
        next = a + b;
        a = b;
        b = next;
        i++;
    } while (i <= n);

    return 0;
}

6) Palindrome String Check - String Functions

#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    string str, rev;
    cout << "Enter a string: ";
    cin >> str;

    rev = str;

}

Experiment 3: Advanced C++ Concepts

1) Complex Number Addition - Operator Overloading

#include <iostream>
using namespace std;

class Complex {
    int real, imag;

public:
    Complex(int r = 0, int i = 0) {
        real = r;
        imag = i;
    }

    // Overload + operator
    Complex operator+(const Complex &c) {
        return Complex(real + c.real, imag + c.imag);
    }

    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
};

int main() {
    Complex c1(3, 4), c2(5, 6);
    Complex c3 = c1 + c2; // uses overloaded +

    cout << "Sum: ";
    c3.display();

    return 0;
}

2) Counter Increment - ++ Operator Overloading

#include <iostream>
using namespace std;

class Counter {
    int value;

public:
    Counter(int v = 0) : value(v) {}

    // Overload prefix ++
    Counter operator++() {
        ++value;
        return *this;
    }

    // Overload postfix ++
    Counter operator++(int) {
        Counter temp = *this;
        value++;
        return temp;
    }

    void display() {
        cout << "Value: " << value << endl;
    }
};

int main() {
    Counter c(5);

    ++c; // prefix
    c.display();

    c++; // postfix
    c.display();

    return 0;
}

6) Word Counter - File Reading

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream file("example.txt");

    if (!file) {
        cout << "File not found!" << endl;
        return 1;
    }

    string word;
    int count = 0;

    while (file >> word) {
        count++;
    }

    cout << "Total words in file: " << count << endl;
    file.close();

    return 0;
}

7) File Copy Operation - File Handling

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream src("source.txt");
    ofstream dest("destination.txt");

    if (!src || !dest) {
        cout << "Error opening file!" << endl;
        return 1;
    }

    char ch;
    while (src.get(ch)) {
        dest.put(ch);
    }

    cout << "File copied successfully!" << endl;

    src.close();
    dest.close();

    return 0;
}
