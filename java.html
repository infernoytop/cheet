2) Object-Oriented Programming - Student Class

class Student {
    private String name;
    private int rollNo;
    private double marks;
    
    public Student(String name, int rollNo, double marks) {
        this.name = name;
        this.rollNo = rollNo;
        this.marks = marks;
    }
    
    public void displayDetails() {
        System.out.println("Name: " + name);
        System.out.println("Roll Number: " + rollNo);
        System.out.println("Marks: " + marks);
    }
    
    public boolean hasPassed() {
        return marks >= 40.0;
    }
}

public class Main {
    public static void main(String[] args) {
        Student student1 = new Student("Sahil", 101, 85.5);
        Student student2 = new Student("Rahul", 102, 35.0);
        
        student1.displayDetails();
        System.out.println("Passed: " + student1.hasPassed());
        System.out.println();
        
        student2.displayDetails();
        System.out.println("Passed: " + student2.hasPassed());
    }
}

3) Rule-Based Expert System - Weather Clothing Suggestions

import java.util.Scanner;

public class ExpertSystem {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.print("Enter weather (sunny, rainy, cold): ");

1) Employee Class - Class Fundamentals

// Save this as Employee.java if your compiler asks for file name
class Employee {
    // Attributes (fields)
    String name;
    int id;
    double salary;
    
    // Constructor
    Employee(String name, int id, double salary) {
        this.name = name;
        this.id = id;
        this.salary = salary;
    }
    
    // Method to display employee details
    void displayDetails() {
        System.out.println("Employee ID: " + id);
        System.out.println("Employee Name: " + name);
        System.out.println("Employee Salary: " + salary);
    }
    
    // Method to give a raise
    void giveRaise(double amount) {
        salary += amount;
        System.out.println(name + " got a raise of " + amount + ". New Salary: " + salary);
    }
    
    // main method inside the same class
    public static void main(String[] args) {
        // Create Employee objects
        Employee e1 = new Employee("Alice", 101, 50000);
        Employee e2 = new Employee("Bob", 102, 60000);
        
        // Display details
        System.out.println("Employee 1 Details:");
        e1.displayDetails();
        System.out.println("\nEmployee 2 Details:");
        e2.displayDetails();
        
        // Give raises
        System.out.println("\nGiving raise...");
        e1.giveRaise(5000);
        e2.giveRaise(7000);
    }
}

2) Student Class - 'this' Keyword Usage

// Demonstrating use of 'this' keyword
class Student {
    // instance variables
    String name;
    int age;
    
    // Constructor with parameters having the same names as instance variables
    Student(String name, int age) {
        // 'this' differentiates instance variables from parameters
        this.name = name;
        this.age = age;
    }
    
    // Method to display student info
    void display() {
        System.out.println("Name: " + name + ", Age: " + age);
    }
    
    // main method to run the program
    public static void main(String[] args) {
        Student s1 = new Student("Alice", 20);
        Student s2 = new Student("Bob", 22);
        
        s1.display();
        s2.display();
    }
}

3) Rectangle Area Calculator - Static Methods and Variables

import java.util.Scanner;

public class RectangleArea {
    // Static variables
    static double length;
    static double width;
    
    // Static method to calculate area
    static double calculateArea() {
        return length * width;
    }
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // Input length and width
        System.out.print("Enter the length of the rectangle: ");
        length = sc.nextDouble();
        
        System.out.print("Enter the width of the rectangle: ");
        width = sc.nextDouble();
        
        // Call static method to calculate area
        double area = calculateArea();
        System.out.println("The area of the rectangle is: " + area);
        
        sc.close();
    }
}

1) Array Manipulation - Finding Largest Element

import java.util.Scanner;

public class LargestInArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        
        int[] arr = new int[n];
        System.out.println("Enter " + n + " elements:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        
        int max = arr[0];
        for (int i = 1; i < n; i++) {
            if (arr[i] > max)
                max = arr[i];
        }
        
        System.out.println("Largest element is: " + max);
        sc.close();
    }
}

Experiment 5

1) Single Inheritance - Person and Student Classes

class Person {
    String name;
    int age;
    
    void displayPerson() {
        System.out.println("Name: " + name + ", Age: " + age);
    }
}

class Student extends Person { // Student inherits from Person
    String course;
    
    void displayStudent() {
        displayPerson(); // call parent method
        System.out.println("Course: " + course);
    }
}

public class Main { // Main class to run
    public static void main(String[] args) {
        Student s = new Student();
        s.name = "Alice";
        s.age = 20;
        s.course = "Computer Science";
        
        s.displayStudent();
    }
}

2) Method Overriding - Animal and Dog Classes

class Animal {
    void sound() {
        System.out.println("Animal makes a sound");
    }
}

class Dog extends Animal { // Dog inherits Animal
    @Override
    void sound() { // overriding parent method
        System.out.println("Dog barks");
    }
}

public class Main {
    public static void main(String[] args) {
        Animal a = new Animal();
        a.sound(); // calls Animal's sound()
        
        Dog d = new Dog();
        d.sound(); // calls Dog's overridden sound()
        
        // Polymorphism: parent reference, child object
        Animal ref = new Dog();
        ref.sound(); // calls Dog's sound() at runtime
    }
}

3) Multiple Inheritance via Interfaces - Vehicle Management System

interface Engine {
    void startEngine();
}

// Second interface
interface Wheels {
    void rotateWheels();
}

// Class implements both interfaces (multiple inheritance)
class Car implements Engine, Wheels {
    public void startEngine() {
        System.out.println("Car engine started");
    }
    
    public void rotateWheels() {
        System.out.println("Car wheels are rotating");
    }
    
    public void showDetails() {
        System.out.println("This is a Car with engine and wheels.");
    }
}

public class Main {
    public static void main(String[] args) {
        Car myCar = new Car();
        
        myCar.showDetails();
        myCar.startEngine();
        myCar.rotateWheels();
    }
}
